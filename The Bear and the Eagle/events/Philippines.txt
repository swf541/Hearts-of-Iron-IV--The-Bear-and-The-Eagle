country_event = {
	id = philippines.1
	hidden = yes
	
	fire_only_once = yes

	trigger = {
		CPH = {
		NOT = {
		controls_state = 872
		}
		has_war_with = PHI
		}
	}
	
	immediate = {
	every_country = { limit = { tag = PHI } annex_country = { target = CPH transfer_troops = no } }

	}

	mean_time_to_happen = { days = 7 }
}
