#Invitation to the League Arabe
country_event = {
	id = egypte.1
	title = {
		text = egypte.1.t1
		trigger = { FROM = { tag = EGY } }
	}
	title = {
		text = egypte.1.t2
		trigger = {
			NOT = { FROM = { tag = EGY } }
		}
	}
	desc = {
		text = egypte.1.d1
		trigger = { FROM = { tag = EGY } }
	}
	desc = {
		text = egypte.1.d2
		trigger = { 
			NOT = { FROM = { tag = EGY } }
		}
	}
	#picture = GFX_report_event_egypte_flag
	
	is_triggered_only = yes
	
	option = {	#We accept
		name = egypte.1.o1
		#set_country_flag = seek_egypte_alliance
		if = {
			limit = { EGY = { is_faction_leader = yes } }
			EGY = { add_to_faction = ROOT }
		}
		if = {
			limit = { EGY = { is_faction_leader = no } }
			random_country = {
				limit = {
					is_in_faction_with = EGY
					is_faction_leader = yes
				}
				add_to_faction = ROOT
			}
		}
		EGY = { country_event = diplomatic_response.1 }
		ai_chance = {
			factor = 10
			modifier = {
				has_government = pan_arabism
				has_government = islamism
				has_government = nationalism
				add = 50
			}

			modifier = {
				factor = 0
				has_government = neutrality
			}
		}
	}
	
	option = {	#We reject
		name = egypte.1.o2
		EGY = { country_event = diplomatic_response.2 }
		ai_chance = {
			factor = 10
			modifier = {
				tag = PAL 
				add = -10
			}
			modifier = {
				tag = JOR
				add = -10
			}
			modifier = {
				tag = SYR
				add = -10
			}
			modifier = {
				tag = LEB
				add = -10
			}
			modifier = {
				tag = SAU
				add = -10
			}

			modifier = {
				tag = IRQ
				add = -10
			}
		}
	}
}

#Surrender when Israel has taken too much land in egypte
country_event = {
	id = egypte.4
	title = egypte.4.t
	desc = egypte.4.d
	picture = GFX_report_event_soviet_tanks_snow
	
	trigger = {
		tag = EGY
		has_war_with = ISR
		surrender_progress > 0.25
		has_civil_war = no
		ISR = { has_completed_focus = ISR_pre_emptive_strike }
	}

	mean_time_to_happen = { days = 1 }

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = egypte.4.a

		ISR = {
			country_event = { id = egypte.5 }
		}
		custom_effect_tooltip = egypte.4.a_tt
		effect_tooltip = {
			ISR = { 
				white_peace = {
					tag = EGY 
					message = EGY_agree_peace 
				} 
			}
			transfer_state = 805
			transfer_state = 453

		}
	}
}
