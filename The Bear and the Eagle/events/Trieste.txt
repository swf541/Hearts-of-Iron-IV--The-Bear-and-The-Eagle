	id = tri.1
	hidden = yes

	fire_only_once = yes

	trigger = {
		date > 1954.10.1
		NOT = { has_global_flag = tri_questions_over }
		date < 1954.10.6
	}

	mean_time_to_happen = {
		days = 7
	}

	immediate = {
		ITA = {
			country_event = ita.1
		}
		YUG = {
			country_event = yug.1
		}
		set_global_flag = tri_questions_over
	}
}
