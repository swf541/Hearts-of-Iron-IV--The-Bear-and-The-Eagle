###########################
# Lebanon Events
###########################

add_namespace = lebanon


#Surrender when Israel has taken too much land in Lebanon
country_event = {
	id = lebanon.1
	title = lebanon.1.t
	desc = lebanon.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	trigger = {
		tag = LEB
		has_war_with = ISR
		surrender_progress > 0.25
		has_civil_war = no
		ISR = { has_completed_focus = ISR_pre_emptive_strike }
	}

	mean_time_to_happen = { days = 0 }

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = lebanon.1.a

		ISR = {
			country_event = { id = lebanon.2 }
		}
		custom_effect_tooltip = lebanon.1.a_tt
		effect_tooltip = {
			ISR = { 
				white_peace = {
					tag = LEB 
					message = LEB_agree_peace 
				} 
			}

		}
	}
}

#Israel response to Lebanon surrender
country_event = {
	id = lebanon.2
	title = lebanon.2.t
	desc = lebanon.2.d
	picture = GFX_report_event_finnish_soldiers_snow
	
	is_triggered_only = yes

	#Accept surrender
	option = {
		name = lebanon.2.a

		white_peace = LEB
	}
}
