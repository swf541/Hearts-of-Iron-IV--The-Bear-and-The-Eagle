###########################
# Jordania Events
###########################

add_namespace = jordania


#Surrender when Israel has taken too much land in Jordania
country_event = {
	id = jordania.1
	title = jordania.1.t
	desc = jordania.1.d
	picture = GFX_report_event_soviet_tanks_snow
	
	trigger = {
		tag = JOR
		has_war_with = ISR
		surrender_progress > 0.25
		has_civil_war = no
		ISR = { has_completed_focus = ISR_pre_emptive_strike }
	}

	mean_time_to_happen = { days = 0 }

	fire_only_once = yes

	#Surrender and offer territory
	option = {
		name = jordania.1.a

		ISR = {
			country_event = { id = jordania.2 }
		}
		custom_effect_tooltip = jordania.1.a_tt
		effect_tooltip = {
			ISR = { 
				white_peace = {
					tag = JOR 
					message = JOR_agree_peace 
				} 
			}
			transfer_state = 332

		}
	}
}

#Israel response to Jordania surrender
country_event = {
	id = jordania.2
	title = jordania.2.t
	desc = jordania.2.d
	picture = GFX_report_event_finnish_soldiers_snow
	
	is_triggered_only = yes

	#Accept surrender
	option = {
		name = jordania.2.a

		white_peace = JOR
		transfer_state = 332
	}
}
